// Understand This
class Car {
  setDriveSound(sound){
  	this.sound = sound;
  }
  
  drive() {
  	return this.sound;
  }
}

const car = new Car();
car.setDriveSound('vroom')
car.drive();

const truck = {
	sound : 'putputput',
  driveMyTruck : car.drive // เรียกเฉยๆ แต่ยังไม่ทำงาน
}

truck.driveMyTruck() // เท่ากับ truck.sound ซึ่งเป็น object ของมันเอง
